# Scout App Implementation Plan (Yutori Scouting + Outreach)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a new `scout` Django app that (1) registers a daily Yutori Scouting API job, receives results via webhook, and dedupes them in Postgres; and (2) runs a daily Celery task that emails each newly-discovered benchmark organizer via EvalAI's existing `send_email()` helper from a dedicated `outreach@eval.ai` SES identity.

**Architecture:** A new top-level Django app at `apps/scout/`. A management command (`scout_create`) registers a recurring Yutori scout pointed at a webhook URL inside this app. Each webhook delivery is persisted raw (`ScoutRun`), then iterated to upsert deduped `ScoutChallenge` rows keyed on a canonical `name|conference|year` string protected by a unique constraint. A separate daily Celery task (`send_daily_outreach`) reads `ScoutChallenge` rows whose `first_seen` is newer than the `LastOutreachRun` watermark and emails each organizer with a non-empty email address. All scouting writes happen inside a single `transaction.atomic()` block; the outreach task uses the watermark for at-least-once semantics.

**Tech Stack:** Django 2.2.20, PostgreSQL (`django.contrib.postgres.fields.JSONField`), Celery (already configured in `evalai/celery.py`), `requests` (already in EvalAI), `responses` (already in `requirements/dev.txt`) for HTTP mocking, `jsonschema` (added in Task 1) for schema validation tests, `pytest` + `pytest-django` (already configured via `pytest.ini`). Outreach reuses `apps/base/utils.py::send_email()` (SendGrid template fetch + Django SES backend).

```python
from __future__ import unicode_literals

from django.contrib.postgres.fields import JSONField
from django.db import models


class Scout(models.Model):
    """The currently-registered Yutori scout. Effectively single-row."""

    scout_id = models.CharField(max_length=64, unique=True)
    query_hash = models.CharField(max_length=64)
    webhook_url = models.URLField(max_length=512)
    yutori_view_url = models.URLField(max_length=512, blank=True)
    created_at = models.DateTimeField(auto_now_add=True)
    paused_at = models.DateTimeField(null=True, blank=True)

    def __str__(self):
        return "Scout(scout_id={}, paused={})".format(
            self.scout_id, self.paused_at is not None
        )


class ScoutRun(models.Model):
    """One row per webhook delivery — audit trail of raw Yutori payloads."""

    received_at = models.DateTimeField(auto_now_add=True, db_index=True)
    raw_payload = JSONField()
    new_challenge_count = models.IntegerField(default=0)
    parse_warnings = JSONField(default=list, blank=True)

    class Meta:
        ordering = ["-received_at"]

    def __str__(self):
        return "ScoutRun(received_at={}, new={})".format(
            self.received_at, self.new_challenge_count
        )


class ScoutChallenge(models.Model):
    """A deduped benchmark/challenge discovered by the scout."""

    benchmark_name = models.CharField(max_length=512)
    conference = models.CharField(max_length=128)
    year = models.IntegerField()
    canonical_key = models.CharField(
        max_length=768, unique=True, db_index=True
    )

    task_area = models.CharField(max_length=128, blank=True)
    official_url = models.URLField(max_length=1024)
    dataset_url = models.URLField(max_length=1024, blank=True)
    organizers = JSONField(default=list)
    evalai_suitable = models.BooleanField()
    evalai_reasoning = models.TextField(blank=True)

    first_seen = models.DateTimeField(auto_now_add=True, db_index=True)
    source_run = models.ForeignKey(
        ScoutRun,
        on_delete=models.SET_NULL,
        null=True,
        related_name="challenges",
    )

    class Meta:
        ordering = ["-first_seen"]

    def __str__(self):
        return "{} @ {} {}".format(
            self.benchmark_name, self.conference, self.year
        )


class LastOutreachRun(models.Model):
    """Single-row watermark consumed by the outreach pipeline (separate spec)."""

    last_run_at = models.DateTimeField()

    def __str__(self):
        return "LastOutreachRun(last_run_at={})".format(self.last_run_at)
```

```python
from scout.dedup import canonical_key


class CanonicalKeyTests(TestCase):
    def test_combines_name_conference_year(self):
        c = {
            "benchmark_name": "ImageNet-21K-P",
            "conference": "NeurIPS",
            "year": 2025,
        }
        self.assertEqual(canonical_key(c), "imagenet-21k-p|neurips|2025")

    def test_lowercases_name_and_conference(self):
        c = {
            "benchmark_name": "FOO",
            "conference": "BAR",
            "year": 2024,
        }
        self.assertEqual(canonical_key(c), "foo|bar|2024")

    def test_strips_whitespace(self):
        c = {
            "benchmark_name": "  imagenet  ",
            "conference": " neurips ",
            "year": 2025,
        }
        self.assertEqual(canonical_key(c), "imagenet|neurips|2025")

    def test_year_is_stringified_as_int(self):
        c = {
            "benchmark_name": "x",
            "conference": "y",
            "year": 2025,
        }
        # The integer 2025 should appear as "2025", not "2025.0" or "2,025"
        self.assertTrue(canonical_key(c).endswith("|2025"))
```

```python
            responses.POST,
            "https://api.yutori.com/v1/scouting/tasks",
            json={
                "id": "scout-uuid-123",
                "query": "test prompt",
                "view_url": "https://platform.yutori.com/scouts/scout-uuid-123",

            status=200,

            query="test prompt",

            webhook_url="https://example.com/api/scout/webhook/abc/",
            output_interval=86400,
```

```python
import requests


YUTORI_API_BASE = "https://api.yutori.com"


class YutoriAPIError(Exception):
    """Raised when the Yutori API returns a non-2xx status."""

    def __init__(self, status_code, body):
        self.status_code = status_code
        self.body = body
        super(YutoriAPIError, self).__init__(
            "Yutori API returned {}: {}".format(status_code, body)
        )


class YutoriClient(object):
    def __init__(self, api_key, base_url=YUTORI_API_BASE):
        self.api_key = api_key
        self.base_url = base_url

    def _headers(self):
        return {
            "X-API-Key": self.api_key,
            "Content-Type": "application/json",
        }

    def _post(self, path, payload):
        url = "{}{}".format(self.base_url, path)
        resp = requests.post(url, json=payload, headers=self._headers())
        if not (200 <= resp.status_code < 300):
            raise YutoriAPIError(resp.status_code, resp.text)
        return resp.json()

    def create_scout(
        self,
        query,
        output_schema,
        webhook_url,
        output_interval=86400,
    ):
        return self._post(
            "/v1/scouting/tasks",
            {
                "query": query,
                "output_schema": output_schema,
                "output_interval": output_interval,
                "webhook_url": webhook_url,
                "webhook_format": "scout",
                "skip_email": True,
            },
        )

    def pause_scout(self, scout_id):
        return self._post(
            "/v1/scouting/tasks/{}/pause".format(scout_id), {}
        )

    def resume_scout(self, scout_id):
        return self._post(
            "/v1/scouting/tasks/{}/resume".format(scout_id), {}
        )
```

Note on pause/resume endpoint paths: the Yutori docs we fetched did not
spell these out explicitly. If integration testing reveals different paths

`test_client.py` together and re-run the test. The contract — header
auth, JSON body, non-2xx raises `YutoriAPIError` — does not change.

```python
import json
import logging

from django.conf import settings
from django.db import transaction
from django.http import HttpResponse, JsonResponse
from django.utils.crypto import constant_time_compare
from django.views.decorators.csrf import csrf_exempt
from django.views.decorators.http import require_POST

from scout.dedup import canonical_key
from scout.models import ScoutChallenge, ScoutRun


logger = logging.getLogger(__name__)


REQUIRED_FIELDS = [
    "benchmark_name",
    "task_area",
    "conference",
    "year",
    "official_url",
    "organizers",
    "evalai_suitable",
    "evalai_reasoning",
]


def _build_defaults(c, run):
    return {
        "benchmark_name": c["benchmark_name"],
        "conference": c["conference"],
        "year": int(c["year"]),
        "task_area": c.get("task_area", ""),
        "official_url": c["official_url"],
        "dataset_url": c.get("dataset_url", "") or "",
        "organizers": c.get("organizers", []) or [],
        "evalai_suitable": bool(c["evalai_suitable"]),
        "evalai_reasoning": c.get("evalai_reasoning", "") or "",
        "source_run": run,
    }


@csrf_exempt
@require_POST
def webhook_receiver(request, token):
    expected = getattr(settings, "YUTORI_WEBHOOK_TOKEN", "") or ""
    if not expected or not constant_time_compare(token, expected):
        return HttpResponse(status=403)

    try:
        body = json.loads(request.body.decode("utf-8"))
    except (ValueError, UnicodeDecodeError):
        return HttpResponse(status=400)

    with transaction.atomic():
        run = ScoutRun.objects.create(raw_payload=body)

        warnings = []
        new_count = 0
        received = 0

        challenges = body.get("challenges")
        if not isinstance(challenges, list):
            warnings.append(
                "Payload is missing 'challenges' list (got {!r})".format(
                    type(body.get("challenges")).__name__
                )
            )
        else:
            for c in challenges:
                received += 1
                missing = [f for f in REQUIRED_FIELDS if f not in c]
                if missing:
                    warnings.append(
                        "Skipped entry missing fields {}: name={!r}".format(
                            missing, c.get("benchmark_name", "?")
                        )
                    )
                    continue
                try:
                    key = canonical_key(c)
                except (KeyError, AttributeError, ValueError) as err:
                    warnings.append(
                        "Could not compute key for entry name={!r}: {}".format(
                            c.get("benchmark_name", "?"), err
                        )
                    )
                    continue
                _, created = ScoutChallenge.objects.get_or_create(
                    canonical_key=key,
                    defaults=_build_defaults(c, run),
                )
                if created:
                    new_count += 1

        run.new_challenge_count = new_count
        run.parse_warnings = warnings
        run.save(update_fields=["new_challenge_count", "parse_warnings"])

    return JsonResponse(
        {"received": received, "new": new_count, "run_id": run.id}
    )
```

```python
class Command(BaseCommand):
    help = "Register a recurring Yutori scout with the current prompt."

    def handle(self, *args, **options):
        api_key = getattr(settings, "YUTORI_API_KEY", "")
        webhook_token = getattr(settings, "YUTORI_WEBHOOK_TOKEN", "")
        public_base = getattr(settings, "SCOUT_PUBLIC_BASE_URL", "")

        if not api_key:
            raise CommandError("YUTORI_API_KEY env var is not set.")
        if not webhook_token:
            raise CommandError("YUTORI_WEBHOOK_TOKEN env var is not set.")
        if not public_base:
            raise CommandError(
                "SCOUT_PUBLIC_BASE_URL env var is not set "
                "(e.g. https://eval.ai or your ngrok URL)."
            )

        live = Scout.objects.filter(paused_at__isnull=True).first()
        if live is not None:
            raise CommandError(
                "A live Scout already exists (scout_id={}). "
                "Pause it first with `manage.py scout_status --pause` "
                "before creating a new one.".format(live.scout_id)
            )

        webhook_url = "{}/api/scout/webhook/{}/".format(
            public_base.rstrip("/"), webhook_token
        )

        client = YutoriClient(api_key=api_key)
        try:
            resp = client.create_scout(
                query=RESEARCH_PROMPT,
                output_schema=OUTPUT_SCHEMA,
                webhook_url=webhook_url,
                output_interval=86400,
            )
        except YutoriAPIError as err:
            raise CommandError("Yutori API call failed: {}".format(err))

        scout_id = resp.get("id")
        if not scout_id:
            raise CommandError(
                "Yutori response missing 'id' field: {!r}".format(resp)
            )

        query_hash = hashlib.sha256(
            RESEARCH_PROMPT.encode("utf-8")
        ).hexdigest()

        scout = Scout.objects.create(
            scout_id=scout_id,
            query_hash=query_hash,
            webhook_url=webhook_url,
            yutori_view_url=resp.get("view_url", "") or "",
        )

        self.stdout.write(
            "Created Scout(scout_id={}). View at: {}".format(
                scout.scout_id, scout.yutori_view_url or "(no view_url)"
            )
        )
```

```python
from django.conf import settings
from django.core.management.base import BaseCommand, CommandError
from django.utils import timezone

from scout.client import YutoriClient, YutoriAPIError
from scout.models import Scout, ScoutRun


class Command(BaseCommand):
    help = (
        "Show, pause, or resume the currently registered Yutori scout. "
        "Default action is to print status."
    )

    def add_arguments(self, parser):
        group = parser.add_mutually_exclusive_group()
        group.add_argument(
            "--pause", action="store_true", help="Pause the active scout."
        )
        group.add_argument(
            "--resume", action="store_true", help="Resume a paused scout."
        )

    def handle(self, *args, **options):
        scout = Scout.objects.order_by("-created_at").first()
        if scout is None:
            self.stdout.write("No Scout registered. Run scout_create first.")
            return

        api_key = getattr(settings, "YUTORI_API_KEY", "")

        if options["pause"]:
            if not api_key:
                raise CommandError("YUTORI_API_KEY env var is not set.")
            client = YutoriClient(api_key=api_key)
            try:
                client.pause_scout(scout.scout_id)
            except YutoriAPIError as err:
                raise CommandError(
                    "Pause failed at Yutori: {}".format(err)
                )
            scout.paused_at = timezone.now()
            scout.save(update_fields=["paused_at"])
            self.stdout.write("Paused Scout(scout_id={}).".format(scout.scout_id))
            return

        if options["resume"]:
            if not api_key:
                raise CommandError("YUTORI_API_KEY env var is not set.")
            client = YutoriClient(api_key=api_key)
            try:
                client.resume_scout(scout.scout_id)
            except YutoriAPIError as err:
                raise CommandError(
                    "Resume failed at Yutori: {}".format(err)
                )
            scout.paused_at = None
            scout.save(update_fields=["paused_at"])
            self.stdout.write(
                "Resumed Scout(scout_id={}).".format(scout.scout_id)
            )
            return

        # Default: status
        self.stdout.write(
            "Scout: id={} created_at={} paused={}".format(
                scout.scout_id,
                scout.created_at.isoformat(),
                scout.paused_at.isoformat() if scout.paused_at else "no",
            )
        )
        self.stdout.write("View URL: {}".format(scout.yutori_view_url or "-"))
        self.stdout.write("\nLast 5 ScoutRuns:")
        recent = list(ScoutRun.objects.all()[:5])
        if not recent:
            self.stdout.write("  (none)")
        for run in recent:
            self.stdout.write(
                "  {}  new={} warnings={}".format(
                    run.received_at.isoformat(),
                    run.new_challenge_count,
                    len(run.parse_warnings or []),
                )
            )
```

```python
from scout.models import ScoutChallenge


def build_template_data(challenge, organizer):
    return {
        "organizer_name": (organizer.get("name") or ""),
        "benchmark_name": challenge.benchmark_name,
        "conference": challenge.conference,
        "year": challenge.year,
        "official_url": challenge.official_url,
        "evalai_pitch": challenge.evalai_reasoning or "",
    }


def iter_new_targets(watermark):
    """Yield (challenge, organizer_dict) pairs for challenges first_seen > watermark.

    Organizers with missing/empty/whitespace-only email are skipped.
    """
    qs = ScoutChallenge.objects.filter(first_seen__gt=watermark).iterator()
    for challenge in qs:
        for organizer in (challenge.organizers or []):
            email = (organizer.get("email") or "").strip()
            if not email:
                continue
            yield challenge, organizer
```

```python
import logging
from datetime import timedelta

from celery import shared_task
from django.conf import settings
from django.utils import timezone

from base.utils import send_email
from scout.models import LastOutreachRun
from scout.outreach import build_template_data, iter_new_targets


logger = logging.getLogger(__name__)


@shared_task
def send_daily_outreach():
    run_started_at = timezone.now()

    watermark, created = LastOutreachRun.objects.get_or_create(
        pk=1,
        defaults={"last_run_at": run_started_at - timedelta(hours=24)},
    )
    if created:
        logger.warning(
            "scout outreach: first run, seeding watermark to now - 24h"
        )

    template_id = settings.SENDGRID_SETTINGS["TEMPLATES"].get(
        "OUTREACH_BENCHMARK_HOSTING"
    )
    sender = settings.OUTREACH_FROM_EMAIL

    sent = 0
    attempted = 0
    for challenge, organizer in iter_new_targets(watermark.last_run_at):
        attempted += 1
        try:
            send_email(
                sender=sender,
                recipient=organizer["email"].strip(),
                template_id=template_id,
                template_data=build_template_data(challenge, organizer),
            )
            sent += 1
        except Exception:
            # send_email already swallows its own exceptions; this is a
            # belt-and-braces guard for anything that escapes.
            logger.exception(
                "scout outreach: send_email raised for recipient=%s",
                organizer.get("email"),
            )

    watermark.last_run_at = run_started_at
    watermark.save(update_fields=["last_run_at"])

    logger.info(
        "scout outreach complete: attempted=%d sent=%d watermark=%s",
        attempted,
        sent,
        run_started_at.isoformat(),
    )
```
